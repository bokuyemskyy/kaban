81 constexpr Bitboard premask(PieceType piece_type, Square square) {
82 assert(piece_type == PieceTypes::BISHOP || piece_type == PieceTypes::ROOK);
83
84 Bitboard edges;
85 if (piece_type == PieceTypes::BISHOP) {
86 edges = (Bitboard::rank(Ranks::R1) | Bitboard::rank(Ranks::R8) | Bitboard::file(Files::FA) |
87 Bitboard::file(Files::FH));
88
89 } else {
90 edges = ((Bitboard::rank(Ranks::R1) | Bitboard::rank(Ranks::R8)) & ~Bitboard::rank(square.rank())) |
91 ((Bitboard::file(Files::FA) | Bitboard::file(Files::FH)) & ~Bitboard::file(square.file()));
92 }
93
94 return Bitboard::slidingAttacks(square, Directions::of(piece_type), Bitboards::ZERO) & ~edges;
95 }

97 template <size_t N>
98 constexpr void generate(PieceType piece_type, std::array<MagicEntry, Squares::count()>& magics,
99 std::array<Bitboard, N>& attacks) {
100 assert(piece_type == PieceTypes::BISHOP || piece_type == PieceTypes::ROOK);
101 static_assert(sizeof(void*) == 8, "Must be 64-bit build");
102 // optimal PRNG seeds of Stockfish
103 // to pseudo randomly generate magics in the shortest time
104 constexpr std::array<uint64_t, 8> seeds = {728, 10316, 55013, 32803, 12281, 15100, 16645, 255};
105
106 size_t attacks_offset = 0;
107
108 for (auto square : Squares::all()) {
109 MagicEntry& magic_entry = magics[square.value()];
110 magic_entry.premask = premask(piece_type, square);
111 magic_entry.shift = Squares::count() - popcount(magic_entry.premask);
112 magic_entry.attacks_offset = attacks_offset;
113
114 std::array<Bitboard, MAX_OCCUPANCIES> occupancies{};
115 std::array<Bitboard, MAX_OCCUPANCIES> reference{};
116 std::array<int, MAX_OCCUPANCIES> epoch{};
117
118 size_t size = 0;
119 int count = 0;
120
121 // use Carry-Rippler method
122 Bitboard possible_occupancy = Bitboards::ZERO;
123 do {
124 occupancies[size] = possible_occupancy;
125 reference[size] = Bitboard::slidingAttacks(square, Directions::of(piece_type), possible_occupancy);
126 size++;
127 possible_occupancy =
128 Bitboard(possible_occupancy.value() - magic_entry.premask.value()) & magic_entry.premask;
129 } while (possible_occupancy != Bitboards::ZERO);
130
131 Prng rng(seeds[square.rank().value()]);
132
133 for (size_t i = 0; i < size;) {
134 for (magic_entry.magic = 0;
135 popcount(Bitboard(magic_entry.magic * magic_entry.premask.value()) >> 56) < 6;) {
136 magic_entry.magic = rng.sparseRand<Magic>();
137 }
138 for (++count, i = 0; i < size; ++i) {
139 size_t index = magic_entry.index(occupancies[i]);
140 if (epoch[index] < count) {
141 epoch[index] = count;
142 attacks[attacks_offset + index] = reference[i];
143 } else if (attacks[attacks_offset + index] != reference[i]) {
144 break;
145 }
146 }
147 }
148
149 attacks_offset += (1ULL << popcount(magic_entry.premask));
150 }
151 }
152};
