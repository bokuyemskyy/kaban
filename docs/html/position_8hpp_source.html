61 template <bool F>
62 [[nodiscard]] bool isLegal() const {
63 if constexpr (F) { // first check, when new position is set. the moves are guaranteed to be legal
64 if (popcount(occupancy<Sides::US>(PieceTypes::KING)) != 1) return false;
65 if (popcount(occupancy<Sides::THEM>(PieceTypes::KING)) != 1) return false;
66 if (popcount(occupancy<Sides::US>(PieceTypes::PAWN)) > 8) return false;
67 if (popcount(occupancy<Sides::THEM>(PieceTypes::PAWN)) > 8) return false;
68 if (popcount(occupancy<Sides::US>()) > 16) return false;
69 if (popcount(occupancy<Sides::THEM>()) > 16) return false;
70 if ((occupancy<Sides::BOTH>(PieceTypes::PAWN) & (Bitboard::rank(Ranks::R1) | Bitboard::rank(Ranks::R8)))
71 .any())
72 return false;
73 }
74
75 Square king = lsb(occupancy<Sides::THEM>(PieceTypes::KING));
76
77 return isSafe<Sides::US>(king);

80 template <Sides Attacker>
81 [[nodiscard]] bool isSafe(Square square) const {
82 assert(Attacker == Sides::US || Attacker == Sides::THEM);
83 if constexpr (Attacker == Sides::US) {
84 if (m_stm == Colors::WHITE) {
85 if ((pawnAttacks<Colors::BLACK>(square) & occupancy<Attacker>(PieceTypes::PAWN)).any()) return false;
86 } else {
87 if ((pawnAttacks<Colors::WHITE>(square) & occupancy<Attacker>(PieceTypes::PAWN)).any()) return false;
88 }
89 } else {
90 if (m_stm == Colors::WHITE) {
91 if ((pawnAttacks<Colors::WHITE>(square) & occupancy<Attacker>(PieceTypes::PAWN)).any()) return false;
92 } else {
93 if ((pawnAttacks<Colors::BLACK>(square) & occupancy<Attacker>(PieceTypes::PAWN)).any()) return false;
94 }
95 }
96 if ((pseudoAttacks<PieceTypes::KING>(square) & occupancy<Attacker>(PieceTypes::KING)).any()) return false;
97 if ((pseudoAttacks<PieceTypes::KNIGHT>(square) & occupancy<Attacker>(PieceTypes::KNIGHT)).any()) return false;
98 if ((pseudoAttacks<PieceTypes::BISHOP>(square) &
99 (occupancy<Attacker>(PieceTypes::BISHOP) | occupancy<Attacker>(PieceTypes::QUEEN)))
100 .any())
101 return false;
102
103 if ((pseudoAttacks<PieceTypes::ROOK>(square) &
104 (occupancy<Attacker>(PieceTypes::ROOK) | occupancy<Attacker>(PieceTypes::QUEEN)))
105 .any())
106 return false;
107
108 return true;
109 }

118

120 move_list = generatePieceMoves<PieceTypes::BISHOP>(occupancy<Sides::US>(PieceTypes::BISHOP), move_list);
121 move_list = generatePieceMoves<PieceTypes::ROOK>(occupancy<Sides::US>(PieceTypes::ROOK), move_list);
122 move_list = generatePieceMoves<PieceTypes::QUEEN>(occupancy<Sides::US>(PieceTypes::QUEEN), move_list);
123 move_list = generatePieceMoves<PieceTypes::KING>(occupancy<Sides::US>(PieceTypes::KING), move_list);
124
125 move_list = generateCastling(move_list);
126 return move_list;
127 } else if constexpr (GT == GenerationTypes::LEGAL) {
128 Move* legal_move_list = move_list;
129
130 move_list = generateMoves<GenerationTypes::ALL>(move_list);
131
132 for (Move* i = legal_move_list; i < move_list; i++) {
133 UndoInfo undo_info = makeMove(*i);
134 if (isLegal<false>()) *legal_move_list++ = *i;
135 unmakeMove(*i, undo_info);
136 }
137
138 return legal_move_list;

172 uint64_t perft(int depth, Move* move_list) {
173 if (depth == 0) return 1;
174
175 Move* move_list_end = generateMoves<GenerationTypes::ALL>(move_list);
176 uint64_t nodes = 0;
177
178 for (Move* i = move_list; i < move_list_end; ++i) {
179 UndoInfo undo = makeMove(*i);
180 if (isLegal<false>()) [[likely]] {
181 nodes += perft(depth - 1, move_list_end);
182 }
183 unmakeMove(*i, undo);
184 }
185
186 return nodes;

251 template <Color C>
252 Move* generatePawnMoves(Bitboard pieces, Move* move_list) const {
253 while (pieces.any()) {
254 Square from = poplsb(pieces);
255
256 Bitboard attacks = pawnAttacks<C>(from) & occupancy<Sides::THEM>();
257 while (attacks.any()) {
258 Square to = poplsb(attacks);
259 if (to.rank() == (C == Colors::WHITE ? Ranks::R8 : Ranks::R1)) {
260 *move_list++ = Move(from, to, MoveFlags::PROMOTION_QUEEN);
261 *move_list++ = Move(from, to, MoveFlags::PROMOTION_ROOK);
262 *move_list++ = Move(from, to, MoveFlags::PROMOTION_BISHOP);
263 *move_list++ = Move(from, to, MoveFlags::PROMOTION_KNIGHT);
264 } else {
265 *move_list++ = Move(from, to);
266 }
267 }
268
269 if (m_en_passant.hasValue()) {
270 if constexpr (C == Colors::WHITE) {
271 if ((pawnAttacks<C>(from) & Bitboard::square(Square(m_en_passant.file(), Ranks::R6))).any())
272 *move_list++ = Move(from, Square(m_en_passant.file(), Ranks::R6), MoveFlags::EN_PASSANT);
273 } else {
274 if ((pawnAttacks<C>(from) & Bitboard::square(Square(m_en_passant.file(), Ranks::R3))).any()) {
275 *move_list++ = Move(from, Square(m_en_passant.file(), Ranks::R3), MoveFlags::EN_PASSANT);
276 }
277 }
278 }
279
280 Square single_push = from.move(C == Colors::WHITE ? Directions::N : Directions::S);
281 if ((Bitboard::square(single_push) & ~occupancy<Sides::BOTH>()).any()) {
282 if (single_push.rank() == (C == Colors::WHITE ? Ranks::R8 : Ranks::R1)) {
283 *move_list++ = Move(from, single_push, MoveFlags::PROMOTION_QUEEN);
284 *move_list++ = Move(from, single_push, MoveFlags::PROMOTION_ROOK);
285 *move_list++ = Move(from, single_push, MoveFlags::PROMOTION_BISHOP);
286 *move_list++ = Move(from, single_push, MoveFlags::PROMOTION_KNIGHT);
287 } else {
288 *move_list++ = Move(from, single_push);
289
290 if constexpr (C == Colors::WHITE) {
291 if (from.rank() == Ranks::R2) {
292 Square double_push = single_push.move(Directions::N);
293 if ((Bitboard::square(double_push) & ~occupancy<Sides::BOTH>()).any())
294 *move_list++ = Move(from, double_push, MoveFlags::PAWN_DOUBLE_PUSH);
295 }
296 } else {
297 if (from.rank() == Ranks::R7) {
298 Square double_push = single_push.move(Directions::S);
299 if ((Bitboard::square(double_push) & ~occupancy<Sides::BOTH>()).any())
300 *move_list++ = Move(from, double_push, MoveFlags::PAWN_DOUBLE_PUSH);
301 }
302 }
303 }
304 }
305 }
306 return move_list;
307 }
